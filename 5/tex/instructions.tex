\section*{Working Equations}

\begin{align}
  \bo{X}
=&\
  \bo{S}^{-\frac{1}{2}}
\\
  D_{\mu\nu}
=&\
  \sum_{i=1}^{n_\text{occ}}
  C_{\mu i}^* C_{\nu i}
\\
  F_{\mu\nu}
=&\
  h_{\mu\nu}
+
  \sum_{\rho\si}
  \ip{\mu\rho||\nu\si}
  D_{\rho\si}
\\
  E
=&\
  \sum_{\mu\nu}
  (h_{\mu\nu}+\tfrac{1}{2}v_{\mu\nu})D_{\mu\nu}
\sp\text{where }
  v_{\mu\nu}
=
  \sum_{\rho\si}
  \ip{\mu\rho||\nu\si}
  D_{\rho\si}
\end{align}
